  (  
     ProductDescription xml

BEGIN  
  INSERT @T  
     SELECT CatalogDescription.query('  
declare namespace PD="http://www.adventure-works.com/schemas/products/description";  
                    //PD:ProductDescription  ')

     WHERE ProductModelID = @ProudctModelID
